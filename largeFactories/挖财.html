function throttle(fn,wait){
  var timer
  return function(){
    var context = this
    var args = arguments
    if(!timer){
      timer = setTimeout(()=>{
        timer =null
        fn.apply(this,args)
      },wait)
    }
  }
}
function debounce(){
  var timer
  return function(){
    var context =this
    var args = arguments

    timer = setTimeout(()=>{
      fn.apply(context,args)
    },wait)
  }
}
