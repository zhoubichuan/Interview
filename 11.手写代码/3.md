## 3.函数节流

```
function throttle(fn, wait) {
    var timer
    return function () {
        var context = this
        var args = arguments
        if (!timer) {
            timer = setTimeout(function () {
                timer = null
                fn.apply(context, args)
            }, wait)
        }
    }
}
window.onresize = throttle(function () {
    console.log(1);
}, 500);
```
