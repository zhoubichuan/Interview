## 8.手动写一个 node 服务器

```
const http = require("http");
const fs = requier("fs");
const server = http.createServer((req, res) => {
  if ((req.url = "/")) {
    const indexFile = fs.createReadStream("./index.html");
    req.writeHead(200, { "content-Type": "text/html;charset=urf8" });
    indexFile.pipe(res);
  }
});
server.listen(3000);
```
