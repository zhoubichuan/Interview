## 7.发布订阅模式（观察者模式）

```js
var event = {
  clientListen: [],
  trigger: function() {
    for (var i = 0; i < this.clientListen.length; i++) {
      var fn = this.clientListen[i];
      fn.apply(this, arguments);
    }
  },
  listen: function(fn) {
    this.clientListen.push(fn);
  }
};
event.listen(function(time) {
  console.log("正式上班时间为：" + time);
});
event.trigger("2018/7");
```
